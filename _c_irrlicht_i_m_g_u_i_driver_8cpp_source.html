508  ImTextureID copyTextureIDFromRawData(irr::video::IVideoDriver * const pIrrDriver, EColorFormat const ColorFormat, irr::u8 * const pPixelData, irr::u32 const Width, irr::u32 const Height)
509  {
510 
511  irr::u32 * pImageData = nullptr;
512  bool IsTempMemoryUsed = false;
513 
514  // decide if we need to translate the color or not
515  switch (ColorFormat)
516  {
517  case ECF_A8R8G8B8:
518  pImageData = reinterpret_cast<irr::u32 *>(pPixelData);
519  IsTempMemoryUsed = false;
520  break;
521 
522  case ECF_R8G8B8A8:
523  pImageData = reinterpret_cast<irr::u32 *>(pPixelData);
524  IsTempMemoryUsed = false;
525 
526  for (irr::u32 X = 0; X < (Width * Height); X++)
527  {
528  irr::video::SColor Color = getColorFromImGuiColor(pImageData[X]);
529  Color.getData(&pImageData[X], irr::video::ECF_A8R8G8B8);
530  }
531 
532  break;
533 
534  case ECF_A8:
535  pImageData = new irr::u32[Width * Height];
536  IsTempMemoryUsed = true;
537 
538  for (irr::u32 X = 0; X < (Width * Height); X++)
539  {
540  // set only Alpha
541  irr::video::SColor Color(pPixelData[X], 255, 255, 255);
542  Color.getData(&pImageData[X], irr::video::ECF_A8R8G8B8);
543  }
544 
545  break;
546 
547  default:
548  LOG_ERROR("Unknown ColorFormat "<<ColorFormat<<"!\n");
549  FASSERT(false);
550  break;
551  }
552 
553  // do not generate mipmaps for font textures
554  pIrrDriver->setTextureCreationFlag(irr::video::ETCF_CREATE_MIP_MAPS, false);
555 
556  // create image object from raw data
557  irr::core::dimension2d<irr::u32> const Size(Width, Height);
558  irr::video::IImage * const pImage = pIrrDriver->createImageFromData(irr::video::ECF_A8R8G8B8, Size, pImageData);
559 
560  // create unique texture name for Irrlicht
561  irr::io::path const TextureName = "IrrIMGUIRaw" + TextureCreationID;
562  TextureCreationID++;
563 
564  // create texture object
565  irr::video::ITexture * const pTexture = pIrrDriver->addTexture(TextureName, pImage);
566  ImTextureID const TextureID = static_cast<ImTextureID>(static_cast<void * const>(pTexture));
567 
568  pIrrDriver->setTextureCreationFlag(irr::video::ETCF_CREATE_MIP_MAPS, true);
569 
570  // delete image object
571  pImage->drop();
572 
573  if (IsTempMemoryUsed)
574  {
575  delete[] pImageData;
576  }
577 
578  LOG_NOTE("{IrrIMGUI-Irr} Create ITexture object from Raw Data. Handle: " << std::hex << TextureID << "\n");
579 
580  return TextureID;
581  }

635  void applyMovingClippingPlaneWorkaround(irr::video::IVideoDriver * const pIrrDriver)
636  {
637  // TODO: Workaround for the wrong clipping plane issue in Irrlicht.
638  // See: http://irrlicht.sourceforge.net/forum/viewtopic.php?f=7&t=50992
639  //
640  // Clipping planes are applied with matrix settings of the last rendered object.
641  // This could be a rotating 3D object, so the clipping plane will rotative with
642  // this object.
643  // To prevent this error, we can draw a very simple and invisible 2D object to the
644  // screen before applying the first clipping plane. Thus the matrix settings are
645  // prepared for 2D and the clipping plane error disappears.
646 
647  irr::u32 const NumberOfVertex = 3;
648  irr::video::S3DVertex * const pDummyVertices = new irr::video::S3DVertex[NumberOfVertex];
649  pDummyVertices[0].Pos = irr::core::vector3df(-20.0f,-20.0f, 0.0f);
650  pDummyVertices[0].Normal = irr::core::vector3df( 0.0f, 0.0f, 1.0f);
651  pDummyVertices[0].Color = irr::video::SColor(0, 255, 0, 0);
652  pDummyVertices[1].Pos = irr::core::vector3df(-10.0f,-20.0f, 0.0f);
653  pDummyVertices[1].Normal = irr::core::vector3df( 0.0f, 0.0f, 1.0f);
654  pDummyVertices[1].Color = irr::video::SColor(0, 255, 0, 0);
655  pDummyVertices[2].Pos = irr::core::vector3df(-20.0f,-10.0f, 0.0f);
